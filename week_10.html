# Cost-Performance (C/P) Analysis
# ===============================

import pandas as pd
import matplotlib.pyplot as plt

# --- 1. Example dataset (you can replace this with your real CSV) ---
data = {
    'product': [
        'Laptop A', 'Laptop B', 'Phone X', 'Phone Y', 'Tablet Z',
        'Headphones M', 'TV P', 'Monitor R', 'Speaker Q', 'Watch S',
        'Camera V', 'Console G', 'Drone H', 'Printer N', 'Projector L'
    ],
    'country': [
        'USA', 'Japan', 'China', 'China', 'Germany',
        'Korea', 'France', 'USA', 'China', 'Italy',
        'Mongolia', 'China', 'Mongolia', 'USA', 'China'
    ],
    'price_usd': [
        1200, 1500, 800, 600, 900,
        150, 1000, 400, 250, 300,
        1100, 500, 950, 200, 700
    ],
    'performance_score': [
        9.0, 9.5, 8.0, 7.5, 8.8,
        6.0, 8.5, 7.8, 6.5, 6.8,
        8.7, 7.0, 9.0, 5.5, 7.5
    ],
    'freight_cost_usd': [
        50, 80, 30, 25, 60,
        10, 40, 15, 12, 20,
        100, 18, 120, 8, 30
    ]
}

df = pd.DataFrame(data)

# --- 2. Compute landed cost (price + freight) ---
df['landed_cost'] = df['price_usd'] + df['freight_cost_usd']

# --- 3. Compute C/P ratio ---
# Assuming performance-to-cost ratio = performance / cost
df['cp_ratio'] = df['performance_score'] / df['landed_cost']

# --- 4. Top 10 products globally ---
top10 = df.sort_values(by='cp_ratio', ascending=False).head(10)
print("Top 10 Highest C/P Products (Global):")
print(top10[['product', 'country', 'cp_ratio']])

# --- 5. Highest C/P product for Mongolia ---
mongolia_df = df[df['country'] == 'Mongolia']
best_mongolia = mongolia_df.sort_values(by='cp_ratio', ascending=False).head(1)
print("\nHighest C/P Product for Mongolia:")
print(best_mongolia[['product', 'landed_cost', 'cp_ratio']])

# --- 6. Visualization ---
plt.figure(figsize=(10,6))
bars = plt.bar(df['product'], df['cp_ratio'], color='lightgray')

# Highlight top 10
for i, p in enumerate(df['product']):
    if p in top10['product'].values:
        bars[i].set_color('orange')

plt.title("Cost-to-Performance (C/P) Ratio by Product")
plt.xlabel("Product")
plt.ylabel("C/P Ratio (Performance per USD)")
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()
