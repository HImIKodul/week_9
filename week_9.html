import pandas as pd
import matplotlib.pyplot as plt

# Load dataset (update the path if needed)
df = pd.read_csv("winemag-data-130k-v2.csv")

# Count wines per country
country_counts = df['country'].value_counts()

# List all 44 countries
print("Total countries:", country_counts.shape[0])
print(country_counts)

# Highlight top 10 countries
top10 = country_counts.head(10)

# Plot all countries
plt.figure(figsize=(14, 6))
bars = plt.bar(country_counts.index, country_counts.values, color='lightgray')

# Highlight top 10
for i, country in enumerate(country_counts.index):
    if country in top10.index:
        bars[i].set_color('orange')

# Label and title
plt.title("Number of Wine Reviews per Country (WineMag Dataset)")
plt.xlabel("Country")
plt.ylabel("Number of Reviews")
plt.xticks(rotation=90)
plt.tight_layout()

# Show plot
plt.show()
